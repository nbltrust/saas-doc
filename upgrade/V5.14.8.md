  enable: true
  url: "https://gcaptcha4.geetest.com"
  captchas:
    "/company/apply:post":
      - "captcha_id(Web)"
      - "captcha_key(Web)"
      - "captcha_id(iOS)"
      - "captcha_key(iOS)"
      - "captcha_id(Android)"
      - "captcha_key(Android)"

      - "captcha_key(Web)"
      - "captcha_id(iOS)"
      - "captcha_key(iOS)"
      - "captcha_id(Android)"
      - "captcha_key(Android)"

      - "captcha_key(Web)"
      - "captcha_id(iOS)"
      - "captcha_key(iOS)"
